## 📊 数据准备

### 创建演示数据脚本
```powershell
# 创建多个用户并执行交易
$users = @("Alice", "Bob", "Charlie", "David", "Eva")

foreach ($name in $users) {
    $id = "demo_$name"
    
    # 创建用户
    $user = @{
        user_id = $id
        username = $name
        initial_cash = 100000
    } | ConvertTo-Json
    
    Invoke-RestMethod -Method Post `
        -Uri http://127.0.0.1:5000/api/v1/trading/user/create `
        -Body $user -ContentType "application/json"
    
    # 随机交易
    $stocks = @("AAPL", "MSFT", "GOOG", "AMZN", "TSLA")
    $stock = $stocks | Get-Random
    $shares = Get-Random -Minimum 5 -Maximum 20
    
    $buy = @{
        user_id = $id
        symbol = $stock
        shares = $shares
    } | ConvertTo-Json
    
    Invoke-RestMethod -Method Post `
        -Uri http://127.0.0.1:5000/api/v1/trading/trade/buy `
        -Body $buy -ContentType "application/json"
}

Write-Output "✅ 演示数据创建完成！"
```
